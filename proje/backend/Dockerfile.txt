# 1. Taban olarak hafif bir Python sürümü seçiyoruz
FROM python:3.10-slim

# 2. Konteyner içindeki çalışma klasörümüzü belirliyoruz
WORKDIR /app

# 3. Önce kütüphane listesini kopyalayıp yüklüyoruz
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 4. Geri kalan tüm Python kodlarını (main.py vb.) konteynere kopyalıyoruz
COPY . .

# 5. Dışarıya açacağımız portu belirtiyoruz
EXPOSE 5001

# 6. Sunucuyu başlatma komutu (0.0.0.0 ile dış bağlantılara izin veriyoruz)
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "5001"]
